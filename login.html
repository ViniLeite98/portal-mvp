/** * auth-check.js * Só verifica se o usuário está logado e redireciona se não estiver. * Não toca no sidebar. * * Uso: * Colocar após supabase.js */ (function() { // Capturar role antes de qualquer coisa var roleExigido = (document.currentScript || {}).getAttribute("data-role") || null; client.auth.getSession().then(function(res) { var session = res.data && res.data.session; // Não logado → vai para login if (!session) { window.location.href = "login.html"; return; } // Verificar permissão se necessário if (roleExigido) { client.from("perfis") .select("role") .eq("id", session.user.id) .single() .then(function(res2) { var role = (res2.data && res2.data.role) || "operadora"; if (roleExigido === "admin" && role !== "admin") { window.location.href = "equipe.html"; } // Salvar role para uso posterior se precisar window.userRole = role; }); } }); })();
